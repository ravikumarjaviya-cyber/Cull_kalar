npm init -y npm install express socket.io const express = require('express'); const app = express(); const http = require('http').createServer(app); const io = require('socket.io')(http); app.get('/', (req, res) => { res.sendFile(__dirname + '/index.html'); }); io.on('connection', (socket) => { console.log('एक यूजर जुड़ गया है'); // मैसेज प्राप्त करना और सबको भेजना socket.on('chat message', (msg) => { io.emit('chat message', msg); }); socket.on('disconnect', () => { console.log('यूजर चला गया'); }); }); http.listen(3000, () => { console.log('सर्वर http://localhost:3000 पर चल रहा है'); }); Live Chat App
