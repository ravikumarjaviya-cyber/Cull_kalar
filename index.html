let turn = "player"; // शुरुआत में आपकी बारी function handleTurn(diceNumber) { if (turn === "player") { console.log("आपने चला: " + diceNumber); // यहाँ आपकी गोटी चलने का कोड आएगा // चाल खत्म होने के बाद कंप्यूटर की बारी turn = "computer"; document.getElementById('message').innerText = "कंप्यूटर की बारी है..."; setTimeout(computerPlay, 1500); // 1.5 सेकंड बाद कंप्यूटर चलेगा } } function computerPlay() { let compNumber = Math.floor(Math.random() * 6) + 1; document.getElementById('dice-box').innerHTML = compNumber; console.log("कंप्यूटर ने चला: " + compNumber); // कंप्यूटर की चाल के बाद वापस आपकी बारी turn = "player"; document.getElementById('message').innerText = "आपकी बारी! पासा फेंकें।"; }
